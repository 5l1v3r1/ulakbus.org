    login function post credentials to API and handles login. If login req returns success then interceptor will redirects to related path.

    authService.login = function (url, credentials) {
    credentials['cmd'] = "do";
    return $http
        .post(RESTURL.url + url, credentials)
        .success(function (data, status, headers, config) {
            //$window.sessionStorage.token = data.token;

            $rootScope.loggedInUser = true;
        })
        .error(function (data, status, headers, config) {
            // Handle login errors here
            data.title = "İşlem başarısız oldu. Lütfen girdiğiniz bilgileri kontrol ediniz."
            return data;
        });
};

    logout function posts logout request to API and redirects to login path

    authService.logout = function () {
    $log.debug("logout");
    return $http.post(RESTURL.url + 'logout', {}).success(function (data) {
        $rootScope.loggedInUser = false;
        $log.debug("loggedout");
        $location.path("/login");
    });
};
